package main

import (
	"fmt"
	"math/big"
)

func isPerfectSquare(num *big.Int) bool {
	sqrt := new(big.Int).Sqrt(num)
	square := new(big.Int).Mul(sqrt, sqrt)
	return square.Cmp(num) == 0
}

func main() {
	numStr := "5" // 设置要检查的整数
	num := new(big.Int)
	num.SetString(numStr, 10)

	if isPerfectSquare(num) {
		fmt.Println("是完全平方数")
	} else {
		fmt.Println("不是完全平方数")
	}
}
